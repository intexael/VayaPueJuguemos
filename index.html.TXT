    <script>
        const timers = {
            1: { remaining: 0, interval: null },
            2: { remaining: 0, interval: null },
            3: { remaining: 0, interval: null }
        };

        function formatTime(seconds) {
            const h = Math.floor(seconds / 3600);
            const m = Math.floor((seconds % 3600) / 60);
            const s = seconds % 60;
            return [h, m, s].map(v => v < 10 ? "0" + v : v).join(":");
        }

        function startTime(id, minutes) {
            // Limpiar si hay uno previo
            clearInterval(timers[id].interval);
            
            // Configurar nuevo tiempo (segundos)
            timers[id].remaining = minutes * 60;
            
            // Actualizar UI
            const card = document.getElementById(`station-${id}`);
            const display = document.getElementById(`display-${id}`);
            card.classList.add('active-glow');
            card.classList.remove('timer-finished');

            timers[id].interval = setInterval(() => {
                if (timers[id].remaining > 0) {
                    timers[id].remaining--;
                    display.innerText = formatTime(timers[id].remaining);
                } else {
                    handleFinish(id);
                }
            }, 1000);
        }

        function handleFinish(id) {
            clearInterval(timers[id].interval);
            const card = document.getElementById(`station-${id}`);
            const alertBox = document.getElementById('alert-box');
            
            card.classList.remove('active-glow');
            card.classList.add('timer-finished');
            
            alertBox.classList.remove('hidden');
            setTimeout(() => {
                alertBox.classList.add('hidden');
            }, 5000);
        }

        function resetTimer(id) {
            clearInterval(timers[id].interval);
            timers[id].remaining = 0;
            
            const card = document.getElementById(`station-${id}`);
            const display = document.getElementById(`display-${id}`);
            
            card.classList.remove('active-glow', 'timer-finished');
            display.innerText = "00:00:00";
        }
    </script>
